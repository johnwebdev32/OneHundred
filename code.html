The Code for One Hundred


                    

                    function getValues(){

                        //get values from the page
                    
                        let startValue = document.getElementById("startValue").value;
                    
                        let endValue = document.getElementById("endValue").value;
                        
                        //parse into Integers
                    
                        startValue = parseInt(startValue);
                    
                        endValue = parseInt(endValue);
                    
                        if (Number.isInteger(startValue) && Number.isInteger(endValue)) {
                           
                            let numbers = generateNumbers(startValue, endValue);
                            
                            displayNumbers(numbers);
                    
                        } else {

                            alert("You must enter integers");
                            
                        }  
                    }
                    
                    //generate numbers from the startValue to the endValue
                    
                    function generateNumbers(startValue, endValue){
                    
                        let numbers = [];
                    
                        //we want to get all numbers from start to end
                    
                        for(let index = startValue; index <= endValue; index++){
                    
                            //this will execute in a loop until index = endValue
                            
                            numbers.push(index);   
                        }
                        return numbers;
                    }
                    
                    //display the numbers 
                    
                    function displayNumbers(numbers){
                    
                        let templateRows = ""; 
                    
                        for (let index = 0; index < numbers.length; index++) {
                    
                            let className = "even";
                    
                            let number = numbers[index];
                    
                            if(number % 2 == 0){
                                className = "even";
                            }
                            else{
                                className = "odd";
                            }
                            
                            templateRows += `${number}`;
                        }
                    
                        document.getElementById("results").innerHTML = templateRows;
                    }

The Code is structured in three functions.

getValues

This function reads the numbers entered by the user and parses them into integers. Then the variables are checked to make sure they are integers. If they are not, the error message is displayed.

generateNumbers

This function declares the array 'numbers' then runs a 'for' loop to push the numbers into the array.

displayNumbers

This function declares a string 'templateRows', then runs a loop to check if the number is odd or even. The numbers are separated based on className = "odd" or "even". They are then displayed and the even numbers are bold using a class in the .css file.
